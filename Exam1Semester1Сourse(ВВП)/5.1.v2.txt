    public static int task5(int[] array) {
        if (array.length == 0) {
            return -1;
        }

        Set<Integer> uniqueElements = new HashSet<>();
        int index = -1;

        for (int i = 0; i < array.length; i++) {
            if (uniqueElements.contains(array[i])) {
            	index = i;
                break;
            }
            uniqueElements.add(array[i]);
            index = i;
        }

        return index;
    }
