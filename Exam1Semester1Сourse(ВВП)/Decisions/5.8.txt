public static int task5(int[] array) {
        for (int i = array.length - 1; i >= 0; i--) {
            int count = 0;
            for (int j = 0; j < array.length; j++) {
                if (array[j] < array[i]) {
                    count++;
                    if (count >= 3) {
                        return i;
                    }
                }
            }
        }
        return -1;
    }
